    ```typescript
    import { createClient } from '@supabase/supabase-js'
    import { NextRequest, NextResponse } from 'next/server'
    import { setUserRole } from '@/lib/auth/roles'
    import { isValidRole } from '@/lib/auth/roles'

    export async function POST(req: NextRequest) {
      const body = await req.json()
      const token: string | undefined = body.token
      const password: string | undefined = body.password

      // 1. Validar input
      if (!token) {
        return NextResponse.json({ error: 'Token ausente' }, { status: 400 })
      }
      if (!password || password.length < 8) {
        return NextResponse.json(
          { error: 'A senha deve ter pelo menos 8 caracteres.' },
          { status: 400 }
        )
      }

      const admin = createClient(
        process.env.NEXT_PUBLIC_SUPABASE_URL!,
        process.env.SUPABASE_SERVICE_ROLE_KEY!,
        { auth: { autoRefreshToken: false, persistSession: false } }
      )

      // 2. Buscar e validar o convite
      const { data: convite, error: convErr } = await admin
        .from('convites')
        .select('id, email, role, loja, tenant_id, usado, expira_em')
        .eq('token', token)
        .single()

      if (convErr || !convite) {
        return NextResponse.json({ error: 'Convite invalido' }, { status: 404 })
      }
      if (convite.usado) {
        return NextResponse.json({ error: 'Este convite ja foi utilizado.' }, { status: 409 })
      }
      if (new Date(convite.expira_em) < new Date()) {
        return NextResponse.json({ error: 'Convite expirado.' }, { status: 410 })
      }
      if (!isValidRole(convite.role)) {
        return NextResponse.json({ error: 'Convite com role invalido' }, { status: 400 })
      }

      // 3. Criar o usuario. Se o convite nao tinha email, gerar um placeholder
      // unico (o usuario foi convidado por link — o gerente passa as credenciais).
      const userEmail = convite.email
        ?? `convite-${convite.id}@dasilva.local`

      const { data: created, error: createErr } = await admin.auth.admin.createUser({
        email: userEmail,
        password,
        email_confirm: true,
        app_metadata: { role: convite.role },
      })
      if (createErr || !created?.user) {
        return NextResponse.json(
          { error: createErr?.message ?? 'Falha ao criar usuario' },
          { status: 400 }
        )
      }
      const newUserId = created.user.id

      // 4. Corrigir o profile criado pelo trigger handle_new_user.
      // O trigger insere o profile com role default 'vendedor' (Pitfall 3).
      // setUserRole reafirma app_metadata.role + profiles.role com o role do convite.
      const roleResult = await setUserRole(admin, newUserId, convite.role)
      if (roleResult.error) {
        return NextResponse.json({ error: roleResult.error }, { status: 400 })
      }

      // Aplicar loja e tenant_id do convite no profile
      const { error: profUpdateErr } = await admin
        .from('profiles')
        .update({ store: convite.loja, tenant_id: convite.tenant_id })
        .eq('id', newUserId)
      if (profUpdateErr) {
        return NextResponse.json({ error: profUpdateErr.message }, { status: 400 })
      }

      // 5. Marcar convite como usado (Pitfall 5: previne reuso do token)
      const { error: usedErr } = await admin
        .from('convites')
        .update({ usado: true })
        .eq('id', convite.id)
      if (usedErr) {
        return NextResponse.json({ error: usedErr.message }, { status: 400 })
      }

      return NextResponse.json({ success: true, email: userEmail })
    }
    ```

    Notas criticas: (1) `app_metadata: { role }` ja e passado no `createUser` E reafirmado por `setUserRole` — redundancia proposital, pois o `createUser` nem sempre persiste app_metadata de forma consistente; (2) a ordem importa — criar user, corrigir profile, e SO ENTAO marcar `usado=true`; se marcar antes e a criacao falhar, o token fica queimado; (3) se o convite nao tem email, gera placeholder `@dasilva.local` — o login real desse usuario sera coordenado pelo gerente.

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Usuario nao-autenticado -> /convite/[token] | Rota publica; o token na URL e a unica credencial |
| Client -> /api/admin/accept-invite | Rota publica; cria conta — seguranca vem 100% da validacao do token |
| Trigger handle_new_user -> profile | O trigger cria o profile com role default; o handler precisa corrigi-lo |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-3-05 | Elevation of Privilege | accept-invite com token expirado | mitigate | Handler checa `new Date(convite.expira_em) < new Date()` -> 410 antes de criar qualquer usuario. Token expirado nunca cria conta |
| T-3-06 | Tampering | Profile criado com role errado (Pitfall 3) | mitigate | Handler chama `setUserRole(admin, newUserId, convite.role)` apos `createUser` — sobrescreve o role default `vendedor` do trigger com o role do convite. app_metadata + profiles ficam sincronizados |
| T-3-17 | Elevation of Privilege | Reuso do token para criar multiplas contas | mitigate | Handler faz `update({ usado: true })` apos criar o usuario; o server component e o handler rejeitam `usado === true`. Token e single-use |
| T-3-18 | Tampering | Client envia role/loja no body para escalar privilegio | mitigate | role, loja e tenant_id vem EXCLUSIVAMENTE da linha `convites` (lida pelo token) — o body so carrega `token` e `password`. Nada de privilegio vem do cliente |
| T-3-19 | Information Disclosure | Card de erro vaza se o token existe ou nao | accept | As mensagens distinguem invalido/expirado/usado — necessario para UX (D-07, UI-SPEC). O token e o segredo; saber que "expirou" nao ajuda um atacante sem o token valido |
| T-3-20 | Denial of Service | Forca bruta de tokens via accept-invite | accept | Token UUID = 122 bits; brute-force inviavel. Supabase aplica rate limit nas Edge/API. Sem mitigacao adicional nesta fase |

ASVS L1: V5.1 (validacao de senha minLength 8 server-side E client-side), V4.1 (privilegio vem do token, nao do cliente), V3.3 (token single-use via flag `usado`), V6.2 (token gerado pelo Postgres).
