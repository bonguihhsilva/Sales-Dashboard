    ```typescript
    // Supabase Edge Function — runtime Deno
    // Envia email de convite via Resend. Deploy: supabase functions deploy send-invite-email --no-verify-jwt

    const RESEND_API_KEY = Deno.env.get('RESEND_API_KEY')

    const corsHeaders = {
      'Access-Control-Allow-Origin': '*',
      'Access-Control-Allow-Headers': 'authorization, x-client-info, apikey, content-type',
      'Access-Control-Allow-Methods': 'POST, OPTIONS',
    }

    Deno.serve(async (req: Request) => {
      // Preflight CORS
      if (req.method === 'OPTIONS') {
        return new Response('ok', { headers: corsHeaders })
      }

      // Sem chave configurada -> responde 200 com sent:false (email e opcional, D-05)
      if (!RESEND_API_KEY) {
        return new Response(
          JSON.stringify({ sent: false, reason: 'RESEND_API_KEY nao configurada' }),
          { status: 200, headers: { ...corsHeaders, 'Content-Type': 'application/json' } },
        )
      }

      let payload: { to?: string; inviteLink?: string; senderName?: string }
      try {
        payload = await req.json()
      } catch {
        return new Response(
          JSON.stringify({ sent: false, reason: 'JSON invalido' }),
          { status: 400, headers: { ...corsHeaders, 'Content-Type': 'application/json' } },
        )
      }

      const { to, inviteLink, senderName } = payload
      if (!to || !inviteLink) {
        return new Response(
          JSON.stringify({ sent: false, reason: 'Campos to e inviteLink sao obrigatorios' }),
          { status: 400, headers: { ...corsHeaders, 'Content-Type': 'application/json' } },
        )
      }

      const html = `
        <div style="font-family: sans-serif; max-width: 480px; margin: 0 auto;">
          <h2>Painel Da Silva</h2>
          <p>${senderName ?? 'Sua equipe'} convidou voce para acessar o Painel de Vendas Da Silva.</p>
          <p>Clique no link abaixo para definir sua senha e acessar o sistema:</p>
          <p><a href="${inviteLink}" style="background:#c8f542;color:#0e0f11;padding:10px 18px;border-radius:8px;text-decoration:none;font-weight:bold;">Definir minha senha</a></p>
          <p style="color:#71717a;font-size:12px;">Este link e valido por 7 dias. Se voce nao esperava este convite, ignore este email.</p>
        </div>
      `.trim()

      const res = await fetch('https://api.resend.com/emails', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'Authorization': `Bearer ${RESEND_API_KEY}`,
        },
        body: JSON.stringify({
          from: 'Painel Da Silva <onboarding@resend.dev>',
          to,
          subject: 'Seu acesso ao Painel Da Silva',
          html,
        }),
      })

      const result = await res.json()
      return new Response(
        JSON.stringify({ sent: res.ok, result }),
        { status: res.ok ? 200 : 502, headers: { ...corsHeaders, 'Content-Type': 'application/json' } },
      )
    })
    ```

    Notas: (1) `from` usa `onboarding@resend.dev` — dominio de teste do Resend que nao exige verificacao; trocar por dominio proprio depois (registrado no `user_setup`); (2) sem `RESEND_API_KEY` a funcao responde 200 com `sent:false` em vez de erro — coerente com D-05; (3) este arquivo NAO e compilado pelo `npm run build` do Next.js (esta em `supabase/functions/`, fora de `src/`) — usa APIs Deno (`Deno.serve`, `Deno.env`); erros de "Deno is not defined" no editor sao esperados e nao afetam o build do app.

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Route handler -> Edge Function | O route handler invoca a Edge Function passando o link de convite |
| Edge Function -> Resend API | A Edge Function envia o email via API HTTP do Resend |
| Edge Function -> chamadores externos | `--no-verify-jwt` torna a funcao chamavel sem JWT |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-3-21 | Information Disclosure | RESEND_API_KEY exposta | mitigate | Chave lida via `Deno.env.get` na Edge Function (secret do Supabase, `supabase secrets set`); nunca embutida no codigo nem prefixada `NEXT_PUBLIC_` |
| T-3-22 | Denial of Service | Edge Function publica (`--no-verify-jwt`) usada para spam de email | accept | A funcao so e invocada com um `inviteLink` valido; envio de spam exigiria conhecer a URL da funcao e o atacante so conseguiria enviar emails com o template fixo de convite. Resend aplica rate limiting. Risco baixo para o contexto (3 lojas) |
| T-3-23 | Spoofing | Email de convite com remetente falsificado | mitigate | `from` fixo (`onboarding@resend.dev` ou dominio verificado no Resend); o cliente nao controla o remetente. SPF/DKIM do Resend reduzem spoofing |
| T-3-24 | Tampering | Falha de email impede criacao do convite | mitigate | Chamada de email isolada em `try/catch`; `emailSent=false` em qualquer falha. O convite e criado e o link e valido independentemente do email (D-05) |
| T-3-25 | Information Disclosure | inviteLink (token) trafega para o Resend | accept | O link contem o token — necessario para o email funcionar. Resend e um provedor confiavel; o token expira em 7 dias e e single-use. Aceitavel |

ASVS L1: V5.1 (Edge Function valida payload `to`/`inviteLink`), V14.2 (segredo via secret manager, nao no codigo), resiliencia (falha de dependencia externa nao quebra o fluxo principal).
